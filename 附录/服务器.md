#### jdk

        JAVA_HOME=/usr/java/jdk1.8.0_131
        JAVA_BIN=/usr/java/jdk1.8.0_131/bin
        PATH=$PATH:$JAVA_BIN
        CLASSPATH=$JAVA_HOME/bin/dt.jar:$JAVA_HOME/lib/tools.jar
        export JAVA_HOME=$JAVA_HOME JAVA_BIN PATH CLASSPATH
